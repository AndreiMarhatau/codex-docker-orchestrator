const createAppTheme = (mode = 'light') => {
  const isDark = mode === 'dark';
 
  return createTheme({
    palette: getPalette(mode, isDark),
    typography: getTypography(),
    shape: {
      borderRadius: 18
    },
    components: getComponents(isDark)
  });
};
 
export default createAppTheme;
